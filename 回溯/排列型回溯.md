# [46. 全排列](https://leetcode.cn/problems/permutations/)
```python
class Solution:
    def permute(self, nums: List[int]) -> List[List[int]]:
        res = []
        n = len(nums)
        visited = set()
        def dfs(path: List[int]):
            if len(path) == n:
                res.append(path)
                return
            for x in nums:
                if x not in visited:
                    visited.add(x)
                    dfs(path+[x])
                    visited.remove(x)
        dfs([])
        return res
```
# [51. N 皇后](https://leetcode.cn/problems/n-queens/)
重点是`is_valid`
```python
class Solution:
    def solveNQueens(self, n: int) -> List[List[str]]:
        # 思路：backtrace, 递归
        if n == 1:
            return [["Q"]]
        board = [["."] * n for _ in range(n)]
        ans = []

        def is_valid(board:List[List[str]], row:int, col:int):
            # 检查同行
            for c in range(n):
                if board[row][c] == "Q":
                    return False
            # 检查列
            for r in range(n):
                if board[r][col] == "Q":
                    return False
            # 检查左上
            r, c = row - 1, col - 1
            while r >= 0 and c >= 0:
                if board[r][c] == "Q":
                    return False
                r -= 1
                c -= 1
            # 检查右上
            r, c = row - 1, col + 1
            while r >= 0 and c < n:
                if board[r][c] == "Q":
                    return False
                r -= 1
                c += 1
            # 检查左下
            r, c = row + 1, col - 1
            while r < n and c >= 0:
                if board[r][c] == "Q":
                    return False
                r += 1
                c -= 1
            # 检查右下
            r, c = row + 1, col + 1
            while r < n and c < n:
                if board[r][c] == "Q":
                    return False
                r += 1
                c += 1
            return True

        def backtrace(board, row:int):
            if row == n:
                board_str = ["".join(row) for row in board]
                ans.append(board_str)
                return
            # 遍历列
            for col in range(n):
                if not is_valid(board, row, col): # try to place queen in (row, col)
                    continue
                # 进入下一行
                board[row][col] = "Q" # 放置皇后
                backtrace(board, row + 1)
                board[row][col] = "." # 回溯
        backtrace(board, 0)
        return ans
```

# [3001. 捕获黑皇后需要的最少移动次数](https://leetcode.cn/problems/minimum-moves-to-capture-the-queen/)
彩蛋题目，帮助熟悉国际象棋
>[!小知识]
>象永远无法到达非同色的点

```python
class Solution:
    def minMovesToCaptureTheQueen(self, a: int, b: int, c: int, d: int, e: int, f: int) -> int:
        # （a,b）white rook (c,d) white bishop，(e,f) black queen
        # we need to find the minimum moves for white rook / white bishop to capture the black queen

        def in_between(l:int, m:int, r:int) -> bool:
            return l<m<r or l>m>r

        # 四种情况直接攻击
        # 1. 车同行，象不在范围内 a==e and (c!=e or not in_between(b,d,f))
        # 2. 车同列，象不在范围内 b==f and (d!=f or not in_between(a,c,e))
        # 3. 象同diagonal，车不在范围内 c+d==e+f and (a+b!=e+f or not in_between(e,a,c))
        # 4. 象同反diagonal，车不在范围内 c-d==e-f and (a-b!=e-f or not in_between(e,a,c))
        
        if a==e and (c!=e or not in_between(b,d,f)) or \
            b==f and (d!=f or not in_between(a,c,e)) or \
            c+d==e+f and (a+b!=e+f or not in_between(e,a,c)) or \
            c-d==e-f and (a-b!=e-f or not in_between(e,a,c)):
            return 1
        
        return 2
```

# [1415. 长度为 n 的开心字符串中字典序第 k 小的字符串](https://leetcode.cn/problems/the-k-th-lexicographical-string-of-all-happy-strings-of-length-n/)
一个 「开心字符串」定义为：

- 仅包含小写字母 `['a', 'b', 'c']`.
- 对所有在 `1` 到 `s.length - 1` 之间的 `i` ，满足 `s[i] != s[i + 1]` （字符串的下标从 1 开始）。

比方说，字符串 **"abc"**，**"ac"，"b"** 和 **"abcbabcbcb"** 都是开心字符串，但是 **"aa"**，**"baa"** 和 **"ababbc"** 都不是开心字符串。

给你两个整数 `n` 和 `k` ，你需要将长度为 `n` 的所有开心字符串按字典序排序。

请你返回排序后的第 k 个开心字符串，如果长度为 `n` 的开心字符串少于 `k` 个，那么请你返回 **空字符串** 。

```python
class Solution:
    def getHappyString(self, n: int, k: int) -> str:
        cnt = 0
        ans = ''
        def backtrack(path: str):
            nonlocal cnt, ans
            if ans:
                return
            
            if len(path) == n:
                cnt += 1
                if cnt == k:
                    ans = path
                return
                
            for c in "abc":
                if not path or c != path[-1]:
                    backtrack(path + c)
        backtrack("")
        return ans
```
