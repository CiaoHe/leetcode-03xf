# [51. N 皇后](https://leetcode.cn/problems/n-queens/)
重点是`is_valid`
```python fold
class Solution:
    def solveNQueens(self, n: int) -> List[List[str]]:
        # 思路：backtrace, 递归
        if n == 1:
            return [["Q"]]
        board = [["."] * n for _ in range(n)]
        ans = []

        def is_valid(board:List[List[str]], row:int, col:int):
            # 检查同行
            for c in range(n):
                if board[row][c] == "Q":
                    return False
            # 检查列
            for r in range(n):
                if board[r][col] == "Q":
                    return False
            # 检查左上
            r, c = row - 1, col - 1
            while r >= 0 and c >= 0:
                if board[r][c] == "Q":
                    return False
                r -= 1
                c -= 1
            # 检查右上
            r, c = row - 1, col + 1
            while r >= 0 and c < n:
                if board[r][c] == "Q":
                    return False
                r -= 1
                c += 1
            # 检查左下
            r, c = row + 1, col - 1
            while r < n and c >= 0:
                if board[r][c] == "Q":
                    return False
                r += 1
                c -= 1
            # 检查右下
            r, c = row + 1, col + 1
            while r < n and c < n:
                if board[r][c] == "Q":
                    return False
                r += 1
                c += 1
            return True

        def backtrace(board, row:int):
            if row == n:
                board_str = ["".join(row) for row in board]
                ans.append(board_str)
                return
            # 遍历列
            for col in range(n):
                if not is_valid(board, row, col): # try to place queen in (row, col)
                    continue
                # 进入下一行
                board[row][col] = "Q" # 放置皇后
                backtrace(board, row + 1)
                board[row][col] = "." # 回溯
        backtrace(board, 0)
        return ans
```

# [46. 全排列](https://leetcode.cn/problems/permutations/)
```python fold
class Solution:
    def permute(self, nums: List[int]) -> List[List[int]]:
        res = []
        n = len(nums)
        visited = set()
        def dfs(path: List[int]):
            if len(path) == n:
                res.append(path)
                return
            for x in nums:
                if x not in visited:
                    visited.add(x)
                    dfs(path+[x])
                    visited.remove(x)
        dfs([])
        return res
```
