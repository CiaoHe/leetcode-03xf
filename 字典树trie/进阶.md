#  [212. 单词搜索 II](https://leetcode.cn/problems/word-search-ii/)
_返回所有二维网格上的单词_ 
```python
class Trie:
    def __init__(self):
        self.children = defaultdict(Trie)
        self.is_end = False

    def insert(self, word: str) -> None:
        node = self
        for c in word:
            node = node.children[c]
        node.is_end = True

    def search(self, word: str) -> bool:
        node = self
        for c in word:
            if c not in node.children:
                return False
            node = node.children[c]
        return node.is_end
    
    def startsWith(self, prefix: str) -> bool:
        node = self
        for c in prefix:
            if c not in node.children:
                return False
            node = node.children[c]
        return True
    
class Solution:
    def findWords(self, board: List[List[str]], words: List[str]) -> List[str]:
        m, n = len(board), len(board[0])
        trie = Trie()
        for word in words:
            trie.insert(word)

        res = []

        def dfs(x, y, node, path) -> None:
            # Base case: If the current node marks the end of a word
            if node.is_end:
                res.append(path)
                node.is_end = False  # To prevent duplicate words
            
            # Temporarily mark the cell as visited
            c = board[x][y]
            board[x][y] = '#'

            # Explore all neighbors
            for dx, dy in [(-1, 0), (1, 0), (0, -1), (0, 1)]:
                nx, ny = x + dx, y + dy
                if 0 <= nx < m and 0 <= ny < n and board[nx][ny] != '#' and board[nx][ny] in node.children:
                    dfs(nx, ny, node.children[board[nx][ny]], path + board[nx][ny])

            # Restore the cell after backtracking
            board[x][y] = c

        # Start DFS from each cell
        for i in range(m):
            for j in range(n):
                c = board[i][j]
                if trie.startsWith(c):  # Start only if the first letter matches
                    dfs(i, j, trie.children[c], c)

        return res
```

# [3093. 最长公共后缀查询](https://leetcode.cn/problems/longest-common-suffix-queries/)

给你两个字符串数组 `wordsContainer` 和 `wordsQuery` 。

对于每个 `wordsQuery[i]` ，你需要从 `wordsContainer` 中找到一个与 `wordsQuery[i]` 有 **最长公共后缀** 的字符串。如果 `wordsContainer` 中有两个或者更多字符串有最长公共后缀，那么答案为长度 **最短** 的。如果有超过两个字符串有 **相同** 最短长度，那么答案为它们在 `wordsContainer` 中出现 **更早** 的一个。

请你返回一个整数数组 `ans` ，其中 `ans[i]`是 `wordsContainer`中与 `wordsQuery[i]` 有 **最长公共后缀** 字符串的下标。

> 插入的时候reverse word, 因为Trie一般用来做longestPrefix

> 每个node还要额外记录best_length 以及 best_index

```python
class TrieNode:
    def __init__(self):
        self.children = defaultdict(TrieNode)
        self.best_length = float('inf')
        self.best_index = float('inf')

class Trie:
    def __init__(self):
        self.root = TrieNode()

    def update_best(self, node: TrieNode, length: int, index: int):
        if length < node.best_length or (length == node.best_length and index < node.best_index):
            node.best_length = length
            node.best_index = index

    def insert(self, word: str, index: int) -> None:
        node = self.root
        self.update_best(node, len(word), index)
        for c in word[::-1]: # Insert in reverse order
            node = node.children[c]
            self.update_best(node, len(word), index)
    
    def query(self, word: str) -> int:
        node = self.root
        for c in word[::-1]: # Query in reverse order
            if c not in node.children:
                break
            node = node.children[c]
        return node.best_index

class Solution:
    def stringIndices(self, wordsContainer: List[str], wordsQuery: List[str]) -> List[int]:
        trie = Trie()
        for i, word in enumerate(wordsContainer):
            trie.insert(word, i)
        result = []
        for word in wordsQuery:
            result.append(trie.query(word))
        return result
```
